# popcorn-nvim

popup-style action menu for Neovim.

## Features

- Make your favorite commands popup
- Structual menu
  - Submenus
  - Separator
- Shortcut
  - Default actions
  - Instant actions by left, right and i
- Dynamic
  - Actions as command strings or functions
  - Menu manipulation functions
  - Name of menu items are generated by evaluating expressions

## Install

### lazy.nvim

```lua
return { "shu-vim/popcorn-nvim" }
```

## Usage

```vim
:Popcorn
```

or

```lua
vim.keymap.set('n', '<C-P>', require("popcorn-nvim").execute)
```

Press `?` to show(toggle) help.

## Setup/Config

### Default config

```lua
return {
    "shu-vim/popcorn-nvim",
    opts = {
        menu = {
            {
                name = 'LSP',
                sub = {
                    {
                        name = 'CodeAction',
                        action = 'lua vim.lsp.buf.code_action()',
                        instant = true,
                        left = function() vim.diagnostic.jump({ count = -1 }) end,
                        right = function() vim.diagnostic.jump({ count = 1 }) end,
                    },
                    {
                        name = 'Diagnostic',
                        action = 'lua vim.diagnostic.open_float({ border = "single" })',
                        left = function() vim.diagnostic.jump({ count = -1 }) end,
                        right = function() vim.diagnostic.jump({ count = 1 }) end,
                    },
                    { name = 'Definition', action = 'lua vim.lsp.buf.definition()' },
                    { name = 'Hover',      action = 'lua vim.lsp.buf.hover()',     default = true },
                    { name = 'Rename',     action = 'lua vim.lsp.buf.rename()' },
                },
            },
            {
                name = 'Color',
                sub = {
                    { name = 'default', action = 'colorschem default', instant = true },
                    { name = '-' },
                    {
                        name = 'Toggle',
                        nameeval = 'printf("Toggle %s", (&background == "dark" ? "light" : "dark"))',
                        action = function()
                            local colors_name = vim.g.colors_name
                            if vim.o.background == 'dark' then
                                vim.o.background = 'light'
                            else
                                vim.o.background = 'dark'
                            end
                            vim.cmd('colorscheme ' .. colors_name)
                            vim.cmd('silent syn on')
                        end,
                        default = true,
                    },
                    { name = '-' },
                    { name = 'SynOn', action = 'syn on' },
                },
            },
            {
                name = 'Tab',
                sub = {
                    { name = 'Only', action = 'tabonly', default = true },
                },
            },
            {
                name = 'Window',
                sub = {
                    {
                        name = 'Alt',
                        action = function() pcall(vim.cmd, 'buffer ' .. vim.fn.bufnr('#')) end,
                        default = true,
                    },
                    { name = 'Split(--)', action = 'split' },
                    { name = 'Split(|)',  action = 'vsplit' },
                },
            },
            { name = '-' },
            { name = 'Buffer',   nameeval = '&fileformat .. " / " .. &fileencoding', skip = true },
            { name = 'Time',     nameeval = 'strftime("%Y-%m-%d %H:%M:%S")',         skip = true },
            { name = '? : help', skip = true },
        },
        HLGroup = 'Comment',
        HLSeparator = 'Comment',
        HLSkip = 'Comment',

        EscToBack = false,
    },
}
```

### Explaning building menu

```lua
return {
    "shu-vim/popcorn-nvim",
    opts = {
        menu = {
            -- simple
            { name = 'item1', action = 'vim.notify("Hello")' },
            -- lua
            { name = 'item1', action = 'lua print("Hello")' },
            -- `name` and `nameeval`
            { name = 'display name/key', nameeval = '"DI".."SP".."LAY"' },
            -- a separator
            { name = '-' },
            -- a parent of submenus
            {
                name = 'parent',
                sub = {
                    { name = 'subitem1', ... },
                    { name = 'subitem2', ... },
                },
            },
            -- action as a function
            {
                name = 'do something',
                action = function(win)
                    -- win: winid of a current window (not a floating window)
                    print('Hello')
                end,
            },
            -- multple execution
            { name = 'multiple', action = {'lua vim.notify("a")', 'lua vim.notify("b")'} },
            -- a parent contains a `default` in its sub
            {
                name = 'press Return to execute `default` action',
                sub = {
                    { name = 'item1', action = 'vim.notify("Hello Default")', default = true },
                },
            },
            -- i keymap can execute action instantly, if <i> appears
            { name = 'item1', action = 'vim.notify("Hello")', instant = true },
            -- h/l keymap can execute action commands
            {
                name = 'Inc/Dec',
                left = [[action 'normal \<C-X>']],
                right = [[action 'normal \<C-A>']],
            }
        },
    }
}
```

## Functions

```lua
require('popcorn-nvim').add(...)
require('popcorn-nvim').remove(...)
require('popcorn-nvim').replace(...)
```

### add

```lua
--- adds a top-level menu item.
--- @param item table : item to be added
--- @param name_elder string :  (optional) `item` is added after an item named `name_elder`
function add(item, name_elder)

-- add last
require('popcorn-nvim').add({name='MyItem', action='lua print("aaa")'})
-- insert after 'LSP'
require('popcorn-nvim').add({name='MyItem', action='lua print("aaa")'}, 'LSP')
```

### remove

```lua
--- Removes a top-level menu item.
--- @param name string : being removed name of an item
function remove(name)

require('popcorn-nvim').remove('MyItem')
```

### replace

```lua
--- replaces a top-level menu item.
--- @param item table : item to be added
--- @param name string : (optional) `item` is added in change of an item named `name`
function replace(item, name)

--require('popcorn-nvim').add({name='MyItem', action='lua print("aaa")'})
require('popcorn-nvim').replace({name='MyNewItem', action='lua print("bbb")'}, 'MyItem')
```
